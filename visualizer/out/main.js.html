/**
 * Represents a Cell
 * 
 */
class Cell {

    /**
     * Indicates default cell color
     *
     * @memberof Cell
     */
    static DEFAULT_COLOR = 'rgba(0, 0, 0, 0.2)'

    /**
     * Creates an instance of Cell.
     * @param {number} x X coordinate in field
     * @param {number} y Y coordinate in field
     * @param {number} value Cell value
     * @param {string} [color={@link Cell.DEFAULT_COLOR}] Cell color (default: {@link Cell.DEFAULT_COLOR})
     * @param {boolean} is_visible Defines is the cell visible on field(default: false)
     */
    constructor(x, y, value, color = Cell.DEFAULT_COLOR, is_visible = false) {
        this.x = x;
        this.y = y;
        this.value = value;
        this.color = color;
        this.is_visible = is_visible;
    }

    /**
     * Paints Cell with a specified color
     *
     * @param {string} color
     */
    paint(color) {
        this.color = color;
    }
}

/**
 * Represents a Player
 *
 */
class Player {
    /**
     * Creates an instance of Player.
     * @param {string} color Players color
     * @param {Cell[]} cells Array of player cells
     * @param {string} name Players name
     */
    constructor(color, cells, name) {
        this.color = color;
        this.cells = cells.map(cell => new Cell(
            cell.x,
            cell.y,
            cell.value,
            this.color,
            true
        ));
        this.name = name;
    }
}

/**
 * Reperesents a Field
 * 
 */
class Field {

    /**
     * Creates an instance of Field.
     * @param {Cell[][]} cells
     * @param {Player[]} players
     */
    constructor(cells, players) {
        /**
         * Width of the field
         * 
         * @type {number}
         */
        this.width = cells[0]?.length || 0;
        /**
         * Height of the field
         * 
         * @type {number}
         */
        this.height = cells.length || 0;
        /**
         * Field represented by cells
         * 
         * @type {number}
         */
        this.cells = cells.map(row => {
            row.map(cell => {
                new Cell(
                    cell.x,
                    cell.y,
                    cell.value,
                    cell.color
                )
            });
        });
        this.players = players.map(player => new Player(
            player.color,
            player.cells,
            player.name
        ));
        this.players.forEach(player => {
            player.cells.forEach(cell => {
                this.cells[cell.x][cell.y] = cell;
            });
        });
    }
}

class Visualization {
    /**
     * Creates an instance of Visualization.

    constructor() {
        this.ticks = []
    }

    /**
     *
     * Adds tick into visualization
     * @param {number} tick
     */
    add_tick(tick) {
        this.ticks.push(tick)
    }

    /**
     *
     * Returns tick of visualization by number
     * @param {number} idx
     * @return {TickView}

    get_tick(idx) {
        return this.ticks[idx]
    }
}

            group.translation.set(x, y);

function setupInteractions(instance) {
    const svg = instance.renderer.domElement;
    var on = false;

    const getShape = (e) => {
        if (e.target.tagName === 'path') {
            return instance.scene.getById(e.target.id);
        }
        return null;
    };

    svg.addEventListener('mouseover', (e) => {
        const shape = getShape(e);
        if (shape) {
            shape.scale = 1.05;
        }
    });

    svg.addEventListener('mouseout', (e) => {
        const shape = getShape(e);
        if (shape) {
            shape.scale = 1.0;
        }
    });

    svg.addEventListener('click', (e) => {
        const shape = getShape(e);
        if (shape) {
            if (shape.stroke === '#f5f6fa') {
                shape.stroke = '#00a8ff';
            } else {
                shape.stroke = '#f5f6fa';
            }
            if (shape.linewidth == 4) {
                shape.linewidth = 1;
            } else {
                shape.linewidth = 4;
            }
            on = ~on;
        }
    });
}
